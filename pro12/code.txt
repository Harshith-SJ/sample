#include<Keypad.h>
const byte rows=4;
const byte cols=4;
char keys[rows][cols]={
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};
byte rowpin[rows]={9,8,7,6};
byte colpin[rows]={5,4,3,2};
Keypad keypad=Keypad(makeKeymap(keys),rowpin,colpin,rows,cols);
void setup(){
  Serial.begin(9600);
  Serial.println("Press any Key");
}
void loop(){
  char key=keypad.getKey();
  if(key){
    Serial.print("Pressed key :");
    Serial.println(key);
  }
}
